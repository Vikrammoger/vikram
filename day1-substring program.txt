//read a string and if a substring of length two appears at both its beginning and end. return a string without the substring at the beginning otherwise return the original string unchanged


class Substring
{
    public static void main(String[] args)
    {
        String s="wt about wt";
        int len=s.length();
        if(len<4)
        {
            System.out.println(s);
            
        }
        else
        {
            if(s.substring(0,2).equals(s.substring(len-2,len)))
            {
                System.out.println(s.substring(2,len));
            }
            else
                System.out.println(s);
                
            
        }
    }
}
